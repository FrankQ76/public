blueprint:
  name: Aqara-Opple Wireless Remote Switch (Double Rocker)
  description: Automate your Aqara-Opple Wireless Remote Switch (Double Rocker) using ZHA events.
  domain: automation
  input:
    aqara_opple_switch:
      name: Select the Aqara Wireless Remote Switch (Double Rocker)
      description: Aqara-Opple Wireless Remote Switch (Double Rocker)
      selector:
        device:
          integration: zha
          manufacturer: LUMI
          entity:
            domain: sensor
            device_class: battery
    button_1:
      name: Button 1
      description: Action to run when button 1 is pressed
      default: []
      selector:
        action:
    button_2:
      name: Button_2
      description: Action to run when button 2 is pressed
      default: []
      selector:
        action:
    button_3:
      name: Button_3
      description: Action to run when button 3 is pressed
      default: []
      selector:
        action:
        
    button_4:
      name: Button_4
      description: Action to run when button 4 is pressed
      default: []
      selector:
        action:
        
    button_3_long_press:
      name: Button_3_Long_Press
      description: Action to run when button 3 is long pressed
      default: []
      selector:
        action:
        
    button_4_long_press:
      name: Button_4_Long_Press
      description: Action to run when button 4 is long pressed
      default: []
      selector:
        action:

mode: restart
max_exceeded: silent

trigger:
  - platform: event
    event_type: zha_event
    event_data:
      device_id: !input aqara_opple_switch

action:
  - variables:
      command: "{{ trigger.event.data.command }}"

  - choose:
      - conditions:
          - "{{ command == 'on' }}"
        sequence: !input button_1
      - conditions:
          - "{{ command == 'off' }}"
        sequence: !input button_2
      - conditions:
          - "{{ command == 'step' && args[1] == '1' }}"
        sequence: !input button_3
      - conditions:
          - "{{ command == 'step' && args[2] == '0' }}"
        sequence: !input button_4
      - conditions:
          - "{{ command == 'step' && args[1] == '1' }}"
        sequence: !input button_3_long_press
      - conditions:
          - "{{ command == 'step' && args[2] == '0' }}"
        sequence: !input button_4_long_press
